jsonschema:"number of results found"`}func SearchTool(ctx context.Context, req *mcp.CallToolRequest, input ToolInput) (    *mcp.CallToolResult,    ToolOutput,    error,) {    // 實作工具邏輯    results := performSearch(ctx, input.Query, input.Limit)        return nil, ToolOutput{        Results: results,        Count:   len(results),    }, nil} // 註冊工具mcp.AddTool(server,     &mcp.Tool{        Name: 錯誤處理始終返回適當的錯誤並使用上下文進行取消：```gofunc MyTool(ctx context.Context, req *mcp.CallToolRequest, input MyInput) (    *mcp.CallToolResult,    MyOutput,    error,) {    // 檢查上下文取消    if ctx.Err() != nil {        return nil, MyOutput{}, ctx.Err()    }        // 返回無效輸入的錯誤    if input.Query == "" {        return nil, MyOutput{}, fmt.Errorf("query cannot be empty")    }        // 執行操作    result, err := performOperation(ctx, input)    if err != nil {        return nil, MyOutput{}, fmt.Errorf("operation failed: %w", err)    }        return nil, result, nil}```## JSON 結構描述標籤使用 `jsonschema`
